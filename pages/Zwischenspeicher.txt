  	/*
	 * get the facebook user id and try to login the user
	 * if no platform user is found, he's redirected to login/register with fb credentials
	 * @param object Facebook Response Object
	*/
	function handleLogin(response){		
		//Try to login with Facebook, if not logged in on platform
		var facebookUserId = response.authResponse.userID;
		
		// look for a user
		jQuery.get('/connect/facebook/getuserid/id/' + facebookUserId, function(data){
			if(data.status==200) { // we found a connected user

				// log in and continue
				jQuery.get('/connect/facebook/loginjs/userid/' + data.id + '/fb_id/' + facebookUserId, function(logindata){
					// login successfull
					if(logindata.status == 200) {
						// reload and happy surfing
	 					window.location.reload();
	 				}
				});
			}
			else{
				// no connected user found, send user to login/register
				FB.api('/me', {fields: 'email'}, function(data) {						
					// no user found -> register a new user / or login
					window.location.href = '/connect/facebook/register/fb_id/'+facebookUserId+'/email/'+data.email;

				});
			}
		});
	}
